# Prototype Design Step Upgrade Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Upgrade the existing `design_contract` step into a user-facing "原型设计" dialogue step that reads confirmed requirement-analysis documents, generates static prototype pages, supports style/audience/platform feedback, and lets users confirm or continue without confirmation.

**Architecture:** Keep the existing step kind `design_contract` as the compatibility key and change its behavior, labels, prompts, artifacts, and UI. The backend generates prototype files inside the Claude attempt directory, registers a manifest and contract as artifacts, serves a safe static preview route, and records whether the prototype was confirmed or skipped. The frontend renders the design step as "原型设计" with an embedded preview, concrete feedback choices when needed, and explicit confirm/continue actions.

**Tech Stack:** Go `factory-server` services and tests, Claude runner integration, existing job/artifact store, React/Vite `sf-portal-mvp`, plain Node script checks, Markdown docs.

---

## Design Decisions Already Locked

- Keep `model.StepDesignContract` and the persisted step kind value `design_contract`; show it as "原型设计" in UI, prompts, docs, and execution graph labels.
- The prototype input is the requirement-analysis output and confirmed requirement documents, not the original raw user prompt.
- Default output is a static prototype page bundle; multi-page is allowed, but the default visible/generated page is the homepage.
- Generate high-fidelity interactive prototypes only when the user explicitly asks for that fidelity.
- Natural-language feedback is accepted first. When feedback is ambiguous, high-impact, or asks for higher fidelity, the system must ask concrete choices.
- Prototype confirmation is optional. If the user confirms, the prototype becomes immutable and a hard downstream constraint. If the user continues without confirmation, downstream steps may use it only as reference context.
- Once confirmed, prototype feedback and regeneration are rejected.

```go
const (
	ClaudeRunReadOnly       ClaudeRunMode = "read_only"
	ClaudeRunWorkspaceWrite ClaudeRunMode = "workspace_write"
	ClaudeRunAttemptWrite   ClaudeRunMode = "attempt_write"

func claudeArgv(codegen bool) []string {
	if codegen {

func claudeArgvForMode(mode ClaudeRunMode) []string {
	argv := []string{"--print", "--verbose", "--output-format", "stream-json"}
	switch mode {
	case ClaudeRunWorkspaceWrite:
		argv = append(argv, "--allowedTools", "Read,Grep,Glob,Edit,MultiEdit,Write,Bash")
	case ClaudeRunAttemptWrite:
		argv = append(argv, "--allowedTools", "Read,Grep,Glob,Edit,MultiEdit,Write", "--disallowedTools", "Bash")
	default:
		argv = append(argv, "--allowedTools", "Read,Grep,Glob", "--disallowedTools", "Edit,MultiEdit,Write,Bash")
	}
	return argv
}
```

```go
func (r *ClaudeRunner) Run(ctx context.Context, ws Workspace, prompt string, inputData any, codegen bool, emit EventEmitter) error {
	mode := ClaudeRunReadOnly
	if codegen {
		mode = ClaudeRunWorkspaceWrite
	}
	return r.RunWithMode(ctx, ws, prompt, inputData, mode, emit)
}

func (r *ClaudeRunner) RunWithMode(ctx context.Context, ws Workspace, prompt string, inputData any, mode ClaudeRunMode, emit EventEmitter) error {
	if emit == nil {
		emit = func(Event) {}
	}
	if err := os.MkdirAll(ws.AttemptDir, 0o755); err != nil {
		return fmt.Errorf("create attempt dir: %w", err)
	}
	promptPath := filepath.Join(ws.AttemptDir, "prompt.md")
	if err := os.WriteFile(promptPath, []byte(prompt), 0o644); err != nil {
		return fmt.Errorf("write prompt: %w", err)
	}
	if inputData != nil {
		b, err := json.MarshalIndent(inputData, "", "  ")
		if err != nil {
			return fmt.Errorf("marshal input: %w", err)
		}
		if err := os.WriteFile(filepath.Join(ws.AttemptDir, "input.json"), b, 0o644); err != nil {
			return fmt.Errorf("write input: %w", err)
		}
	}

	argv := claudeArgvForMode(mode)
	cmd := exec.CommandContext(ctx, r.binary(), argv...)
	cmd.Dir = runDirForMode(ws, mode)
	cmd.Env = os.Environ()
	cmd.Stdin = strings.NewReader(prompt)

	emit(Event{Stage: stageNameForMode(mode), Message: "claude runner started"})
	return r.runStream(ctx, cmd, ws.AttemptDir, emit)
}

func runDirForMode(ws Workspace, mode ClaudeRunMode) string {
	switch mode {
	case ClaudeRunWorkspaceWrite:
		if ws.WorkDir != "" {
			return ws.WorkDir
		}
		return ws.Root
	case ClaudeRunAttemptWrite:
		return ws.AttemptDir
	default:
		return ws.Root
	}
}

func stageNameForMode(mode ClaudeRunMode) string {
	switch mode {
	case ClaudeRunWorkspaceWrite:
		return "claude_codegen"
	case ClaudeRunAttemptWrite:
		return "claude_attempt_write"
	default:
		return "claude_readonly"
	}
}
```

```go
package executor

import (
	"encoding/json"
	"fmt"
	"os"
	"path/filepath"
	"strings"

	"factory-server/internal/model"
)

type prototypeDesignOutput struct {
	Summary      string `json:"summary"`
	PrototypeDir string `json:"prototype_dir"`
	Manifest     string `json:"manifest"`
	Contract     string `json:"contract"`
}

type validatedPrototypeOutput struct {
	Output       prototypeDesignOutput
	Manifest     model.PrototypeManifest
	Contract     model.PrototypeContract
	PrototypeDir string
	ManifestPath string
	ContractPath string
	DefaultPath  string
}

func validatePrototypeDesignOutput(outputPath string, attemptDir string) (validatedPrototypeOutput, error) {
	var out prototypeDesignOutput
	if err := readJSONFile(outputPath, &out); err != nil {
		return validatedPrototypeOutput{}, fmt.Errorf("read prototype output %s: %w", outputPath, err)
	}
	if strings.TrimSpace(out.PrototypeDir) == "" {
		return validatedPrototypeOutput{}, fmt.Errorf("prototype output missing prototype_dir")
	}
	prototypeDir, err := safeAttemptPath(attemptDir, out.PrototypeDir)
	if err != nil {
		return validatedPrototypeOutput{}, fmt.Errorf("prototype_dir: %w", err)
	}
	manifestPath, err := safeAttemptPath(attemptDir, out.Manifest)
	if err != nil {
		return validatedPrototypeOutput{}, fmt.Errorf("manifest: %w", err)
	}
	contractPath, err := safeAttemptPath(attemptDir, out.Contract)
	if err != nil {
		return validatedPrototypeOutput{}, fmt.Errorf("contract: %w", err)
	}

	var manifest model.PrototypeManifest
	if err := readJSONFile(manifestPath, &manifest); err != nil {
		return validatedPrototypeOutput{}, fmt.Errorf("read prototype manifest: %w", err)
	}
	var contract model.PrototypeContract
	if err := readJSONFile(contractPath, &contract); err != nil {
		return validatedPrototypeOutput{}, fmt.Errorf("read prototype contract: %w", err)
	}
	if err := validatePrototypeManifest(manifest); err != nil {
		return validatedPrototypeOutput{}, err
	}
	defaultPath, err := safeAttemptPath(prototypeDir, manifest.DefaultPage)
	if err != nil {
		return validatedPrototypeOutput{}, fmt.Errorf("default page: %w", err)
	}
	if _, err := os.Stat(defaultPath); err != nil {
		return validatedPrototypeOutput{}, fmt.Errorf("default page %s: %w", manifest.DefaultPage, err)
	}
	return validatedPrototypeOutput{
		Output:       out,
		Manifest:     manifest,
		Contract:     contract,
		PrototypeDir: prototypeDir,
		ManifestPath: manifestPath,
		ContractPath: contractPath,
		DefaultPath:  defaultPath,
	}, nil
}

func validatePrototypeManifest(manifest model.PrototypeManifest) error {
	if strings.TrimSpace(manifest.DefaultPage) == "" {
		return fmt.Errorf("prototype manifest missing default_page")
	}
	if len(manifest.Pages) == 0 {
		return fmt.Errorf("prototype manifest must include at least one page")
	}
	for _, page := range manifest.Pages {
		if strings.TrimSpace(page.ID) == "" || strings.TrimSpace(page.Path) == "" {
			return fmt.Errorf("prototype manifest page must include id and path")
		}
	}
	return nil
}

func safeAttemptPath(root string, rel string) (string, error) {
	if filepath.IsAbs(rel) {
		return "", fmt.Errorf("absolute paths are not allowed: %s", rel)
	}
	clean := filepath.Clean(rel)
	if clean == "." || strings.HasPrefix(clean, "..") {
		return "", fmt.Errorf("path escapes attempt directory: %s", rel)
	}
	rootAbs, err := filepath.Abs(root)
	if err != nil {
		return "", err
	}
	joined, err := filepath.Abs(filepath.Join(rootAbs, clean))
	if err != nil {
		return "", err
	}
	if joined != rootAbs && !strings.HasPrefix(joined, rootAbs+string(os.PathSeparator)) {
		return "", fmt.Errorf("path escapes attempt directory: %s", rel)
	}
	return joined, nil
}

func readJSONFile(path string, target any) error {
	b, err := os.ReadFile(path)
	if err != nil {
		return err
	}
	if err := json.Unmarshal(b, target); err != nil {
		return err
	}
	return nil
}
```

```go
func (r *artifactRegistrar) registerExistingFile(ctx context.Context, kind string, sourcePath string, title string, metadata map[string]any) (*model.Artifact, error) {
	if r == nil || r.store == nil {
		return nil, nil
	}
	info, err := os.Stat(sourcePath)
	if err != nil {
		return nil, fmt.Errorf("stat artifact source: %w", err)
	}
	if info.IsDir() {
		return nil, fmt.Errorf("artifact source is a directory: %s", sourcePath)
	}
	return r.store.CreateArtifact(ctx, model.Artifact{
		JobID:     r.job.ID,
		StepID:    r.step.ID,
		Attempt:   r.step.Attempt,
		Kind:      kind,
		Title:     title,
		Path:      sourcePath,
		Bytes:     info.Size(),
		Metadata:  metadata,
		CreatedAt: time.Now().UTC(),
	})
}
```

```go
func safePrototypeFile(root string, rel string) (string, error) {
	if filepath.IsAbs(rel) {
		return "", errBadPrototypePath
	}
	clean := filepath.Clean(rel)
	if clean == "." || strings.HasPrefix(clean, "..") {
		return "", errBadPrototypePath
	}
	rootAbs, err := filepath.Abs(root)
	if err != nil {
		return "", err
	}
	joined, err := filepath.Abs(filepath.Join(rootAbs, clean))
	if err != nil {
		return "", err
	}
	if joined != rootAbs && !strings.HasPrefix(joined, rootAbs+string(filepath.Separator)) {
		return "", errBadPrototypePath
	}
	return joined, nil
}
```

```go
func prototypePromptContext(artifacts []model.Artifact) string {
	var latest *model.Artifact
	for i := range artifacts {
		artifact := artifacts[i]
		if artifact.Kind == "prototype_contract" {
			latest = &artifact
		}
	}
	if latest == nil || latest.Path == "" {

	}
	var contract model.PrototypeContract
	if err := readJSONFile(latest.Path, &contract); err != nil {
		return ""

	constraintLevel := "参考"

	}
	return fmt.Sprintf(`[prototype_design_context]
约束级别：%s
风格：%s
目标用户：%s
目标平台：%s
保真度：%s
约束：%s
[/prototype_design_context]`, constraintLevel, contract.Style, contract.Audience, contract.Platform, contract.Fidelity, strings.Join(contract.Constraints, "；"))
}
```
